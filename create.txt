CREATE MULTISET TABLE TERADATAPOINT.STUDENT ,NO FALLBACK ,
     NO BEFORE JOURNAL,
     NO AFTER JOURNAL,
     CHECKSUM = DEFAULT,
     DEFAULT MERGEBLOCKRATIO
     (
      Roll_No CHAR(10)
      ,First_Name VARCHAR(20)
      ,Last_Name VARCHAR(20)
      ,DOB DATE format 'YYYY-MM-DD'
     )
PRIMARY INDEX (Roll_No);

/*Section 4*/
.Record 2;
.SET RECORD VARTEXT '|';
